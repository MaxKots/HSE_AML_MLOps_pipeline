from __future__ import annotations

from datetime import datetime, timedelta
from pprint import pformat

from airflow.decorators import dag, task
from airflow.operators.empty import EmptyOperator
from airflow.operators.trigger_dagrun import TriggerDagRunOperator

from src.pipelines.drift_pipeline import run_drift_pipeline
from src.utils.logger import get_logger

logger = get_logger(__name__)


default_args = {
    "owner": "aml-mlops",
    "depends_on_past": False,
    "email_on_failure": False,
    "email_on_retry": False,
    "retries": 2,
    "retry_delay": timedelta(minutes=5),
}


@dag(
    dag_id="aml_drift_dag",
    description="Проверка дрейфа данных и запуск переобучения при необходимости",
    default_args=default_args,
    start_date=datetime(2025, 1, 1),
    schedule="@daily",
    catchup=False,
    tags=["aml", "drift", "mlops"],
)
def aml_drift_dag():
    start = EmptyOperator(task_id="start")
    no_drift = EmptyOperator(task_id="no_drift_detected")
    finish = EmptyOperator(task_id="finish")

    @task(task_id="run_drift_check")
    def run_drift_check_task(current_dataset_name: str = "variant_1") -> dict:
        logger.info(f"Запуск drift-check для датасета '{current_dataset_name}'")

        summary = run_drift_pipeline(
            reference_dataset_name="base",
            current_dataset_name=current_dataset_name,
        )

        logger.info("Drift-check завершён")
        logger.info(pformat(summary))

        return summary

    @task.branch(task_id="branch_on_drift")
    def branch_on_drift_task(drift_summary: dict) -> str:
        drift_detected = bool(drift_summary["drift_detected"])

        if drift_detected:
            logger.info("Обнаружен дрейф, переход в ветку retraining")
            return "trigger_retraining"

        logger.info("Дрейф не обнаружен, завершение без переобучения")
        return "no_drift_detected"

    trigger_retraining = TriggerDagRunOperator(
        task_id="trigger_retraining",
        trigger_dag_id="aml_training_dag",
        conf={"dataset_name": "base"},
        wait_for_completion=False,
        reset_dag_run=True,
    )

    drift_summary = run_drift_check_task()
    branch_decision = branch_on_drift_task(drift_summary)

    start >> drift_summary >> branch_decision
    branch_decision >> trigger_retraining >> finish
    branch_decision >> no_drift >> finish

from __future__ import annotations

from datetime import datetime, timedelta
from pprint import pformat

from airflow.decorators import dag, task
from airflow.operators.empty import EmptyOperator
from airflow.operators.python import get_current_context

from src.pipelines.training_pipeline import run_training_pipeline
from src.utils.logger import get_logger

logger = get_logger(__name__)


default_args = {
    "owner": "aml-mlops",
    "depends_on_past": False,
    "email_on_failure": False,
    "email_on_retry": False,
    "retries": 2,
    "retry_delay": timedelta(minutes=5),
}


@dag(
    dag_id="aml_training_dag",
    description="Обучение AML-модели на базовом датасете",
    default_args=default_args,
    start_date=datetime(2025, 1, 1),
    schedule=None,
    catchup=False,
    tags=["aml", "training", "mlops"],
)
def aml_training_dag():
    start = EmptyOperator(task_id="start")
    finish = EmptyOperator(task_id="finish")

    @task(task_id="read_runtime_config")
    def read_runtime_config_task() -> dict:
        context = get_current_context()
        dag_run = context.get("dag_run")

        conf = dag_run.conf if dag_run and dag_run.conf else {}
        dataset_name = conf.get("dataset_name", "base")

        runtime_config = {
            "dataset_name": dataset_name,
        }

        logger.info(f"Runtime-конфигурация прочитана: {runtime_config}")
        return runtime_config

    @task(task_id="run_training")
    def run_training_task(runtime_config: dict) -> dict:
        dataset_name = runtime_config["dataset_name"]

        logger.info(f"Запуск задачи обучения для датасета '{dataset_name}'")

        summary = run_training_pipeline(dataset_name=dataset_name)

        logger.info("Задача обучения завершена")
        logger.info(pformat(summary))

        return summary

    runtime_config = read_runtime_config_task()
    training_summary = run_training_task(runtime_config)

    start >> runtime_config >> training_summary >> finish

@st.cache_data
def load_local_dataset(dataset_name: str) -> pd.DataFrame:
    dataset_map = {
        "Base": settings.base_dataset,
        "Variant I": settings.drift_dataset_1,
        "Variant II": settings.drift_dataset_2,
    }

    path = dataset_map[dataset_name]
    return load_dataframe(path)


def check_api_health() -> dict:
    try:
        response = requests.get(f"{API_URL}/health", timeout=10)
        response.raise_for_status()
        return response.json()
    except Exception as exc:
        return {
            "status": "error",
            "is_model_loaded": False,
            "loaded_model_type": None,
            "error": str(exc),
        }


def prepare_transactions_payload(df: pd.DataFrame) -> list[dict]:
    working_df = df.drop(columns=["fraud_bool"], errors="ignore").copy()

    for column in INTEGER_COLUMNS:
        if column in working_df.columns:
            working_df[column] = pd.to_numeric(working_df[column], errors="coerce").round().astype("Int64")

    for column in FLOAT_COLUMNS:
        if column in working_df.columns:
            working_df[column] = pd.to_numeric(working_df[column], errors="coerce").astype(float)

    for column in STRING_COLUMNS:
        if column in working_df.columns:
            working_df[column] = working_df[column].astype(str)

    records = working_df.where(pd.notnull(working_df), None).to_dict(orient="records")
    return records


def score_batch(df: pd.DataFrame) -> list[dict]:
    payload = {
        "transactions": prepare_transactions_payload(df)
    }

    response = requests.post(
        f"{API_URL}/predict_batch",
        json=payload,
        timeout=120,
    )

    if not response.ok:
        raise ValueError(f"API error {response.status_code}: {response.text}")

    return response.json()["items"]


def load_drift_summary() -> list[dict]:
    reports_dir = get_artifacts_dir() / "reports"
    if not reports_dir.exists():
        return []

    summaries = []
    for path in sorted(reports_dir.glob("drift_*.yaml")):
        try:
            report = read_yaml(path)
            metrics = report["metrics"][0]["result"]

            summaries.append(
                {
                    "report_file": path.name,
                    "dataset_drift": metrics.get("dataset_drift"),
                    "share_of_drifted_columns": metrics.get("share_of_drifted_columns"),
                    "number_of_drifted_columns": metrics.get("number_of_drifted_columns"),
                    "number_of_columns": metrics.get("number_of_columns"),
                }
            )
        except Exception:
            continue

    return summaries


def recommendation_to_label(recommendation: str) -> str:
    mapping = {
        "red": "🔴 block",
        "yellow": "🟡 review",
        "green": "🟢 pass",
    }
    return mapping.get(recommendation, recommendation)

### Важно: ``AIRFLOW_UID``

Airflow в Docker работает с локально примонтированными директориями проекта, поэтому ему нужен корректный ``AIRFLOW_UID``.

В файле ``.env`` должно быть указано, например:

```env
AIRFLOW_UID=1000
```

Проверить свой UID можно так:

```bash
id -u
```

Если команда возвращает не ``1000``, замени значение ``AIRFLOW_UID`` в ``.env`` на своё.

Это нужно для того, чтобы:

- контейнеры Airflow корректно запускались;
- логи и артефакты создавались с ожидаемыми правами;
- не возникали ошибки доступа к файлам на хосте;
- не приходилось вручную исправлять владельца файлов через ``chown``.

Если ``AIRFLOW_UID`` задан неверно или не задан вообще, возможны проблемы со стартом Airflow и ошибки прав доступа при записи в ``logs/`, ``data/processed/` и ``artifacts/`.

## 8. Частые проблемы

### Проблема: database `mlflow` does not exist

Проверить список баз:

```bash
docker-compose exec postgres psql -U aml_user -d postgres -c "\l"
```

Если базы нет, значит init-скрипт не выполнился, либо volume PostgreSQL уже существовал до добавления SQL-файла.

Решение: сделать чистый сброс PostgreSQL через `docker-compose down -v` и поднять заново.

---

### Проблема: Airflow пишет `You need to upgrade the database`

Причина: Airflow metadata DB не инициализирована или используется не та база.

Нужно, чтобы Airflow использовал отдельную БД `airflow`, а MLflow — отдельную БД `mlflow`.

После исправления конфигурации нужно поднять инфраструктуру заново.

---

### Проблема: `Permission denied` в `artifacts`

Причина: часть файлов создана контейнером под другим пользователем.

Быстрое решение:

```bash
sudo rm -rf artifacts
mkdir -p artifacts/metrics artifacts/models artifacts/reports artifacts/data artifacts/predictions artifacts/shap
```

---

### Проблема: MLflow долго стартует

Если внутри контейнера каждый раз выполняется `pip install`, запуск может быть медленным.

Это не критическая ошибка, но лучше вынести установку зависимостей в отдельный Dockerfile для сервиса MLflow.

  esac
}

should_include_file() {
  local file="$1"
  local base lower

  base="$(basename "$file")"
  lower="$(printf '%s' "$base" | tr '[:upper:]' '[:lower:]')"

  case "$lower" in
    dockerfile|makefile|readme|readme.md|license|license.md|requirements.txt|\
pyproject.toml|poetry.lock|pipfile|pipfile.lock|setup.py|setup.cfg|tox.ini|\
.gitignore|.dockerignore|.env|.env.example|docker-compose.yml|docker-compose.yaml)
      return 0
      ;;
  esac

  case "$lower" in
    *.py|*.sh|*.bash|*.zsh|*.md|*.txt|*.yml|*.yaml|*.json|*.toml|*.ini|*.cfg|*.conf|*.sql|*.xml)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

is_text_file() {
  local file="$1"

  if ! command -v file >/dev/null 2>&1; then
    return 0
  fi

  if file --mime "$file" | grep -q 'charset=binary'; then
    return 1
  fi

  return 0
}

get_file_size() {
  local file="$1"

  if stat -c%s "$file" >/dev/null 2>&1; then
    stat -c%s "$file"
    return 0
  fi

  if stat -f%z "$file" >/dev/null 2>&1; then
    stat -f%z "$file"
    return 0
  fi

  wc -c < "$file"
}

contains_embedded_binary_blob() {
  local file="$1"

  if grep -qE 'data:(font|image|application)/[^;]+;base64,' "$file" 2>/dev/null; then
    return 0
  fi

  if awk 'length($0) > 5000 { found=1; exit } END { exit !found }' "$file" 2>/dev/null; then
    return 0
  fi

  return 1
}

@dag(
    dag_id="aml_drift_dag",
    description="Проверка дрейфа данных и запуск переобучения при необходимости",
    default_args=default_args,
    start_date=datetime(2025, 1, 1),
    schedule="@daily",
    catchup=False,
    tags=["aml", "drift", "mlops"],
)
def aml_drift_dag():
    start = EmptyOperator(task_id="start")
    no_drift = EmptyOperator(task_id="no_drift_detected")
    finish = EmptyOperator(task_id="finish")

    @task(task_id="run_drift_check")
    def run_drift_check_task(current_dataset_name: str = "variant_1") -> dict:

        summary = run_drift_pipeline(
            reference_dataset_name="base",
            current_dataset_name=current_dataset_name,
        )

        logger.info("Drift-check завершён")
        logger.info(pformat(summary))

        return summary

    @task.branch(task_id="branch_on_drift")
    def branch_on_drift_task(drift_summary: dict) -> str:
        drift_detected = bool(drift_summary["drift_detected"])

        if drift_detected:
            logger.info("Обнаружен дрейф, переход в ветку retraining")
            return "trigger_retraining"

        logger.info("Дрейф не обнаружен, завершение без переобучения")
        return "no_drift_detected"

    trigger_retraining = TriggerDagRunOperator(
        task_id="trigger_retraining",
        trigger_dag_id="aml_training_dag",
        conf={"dataset_name": "base"},
        wait_for_completion=False,
        reset_dag_run=True,
    )

    drift_summary = run_drift_check_task()
    branch_decision = branch_on_drift_task(drift_summary)

    start >> drift_summary >> branch_decision
    branch_decision >> trigger_retraining >> finish
    branch_decision >> no_drift >> finish

def check_api_health() -> dict:
    try:
        response = requests.get(f"{API_URL}/health", timeout=10)
        response.raise_for_status()
        return response.json()
    except Exception as exc:
        return {
            "status": "error",
            "is_model_loaded": False,
            "loaded_model_type": None,
            "error": str(exc),
        }


def prepare_transactions_payload(df: pd.DataFrame) -> list[dict]:
    working_df = df.drop(columns=["fraud_bool"], errors="ignore").copy()

    for column in INTEGER_COLUMNS:
        if column in working_df.columns:
            working_df[column] = pd.to_numeric(working_df[column], errors="coerce").round().astype("Int64")

    for column in FLOAT_COLUMNS:
        if column in working_df.columns:
            working_df[column] = pd.to_numeric(working_df[column], errors="coerce").astype(float)

    for column in STRING_COLUMNS:
        if column in working_df.columns:
            working_df[column] = working_df[column].astype(str)

    records = working_df.where(pd.notnull(working_df), None).to_dict(orient="records")
    return records


def score_batch(df: pd.DataFrame) -> list[dict]:
    payload = {
        "transactions": prepare_transactions_payload(df)
    }

    response = requests.post(
        f"{API_URL}/predict_batch",
        json=payload,
        timeout=120,
    )

    if not response.ok:
        raise ValueError(f"API error {response.status_code}: {response.text}")

    return response.json()["items"]

            summaries.append(

                    "report_file": path.name,
                    "dataset_drift": metrics.get("dataset_drift"),
                    "share_of_drifted_columns": metrics.get("share_of_drifted_columns"),
                    "number_of_drifted_columns": metrics.get("number_of_drifted_columns"),
                    "number_of_columns": metrics.get("number_of_columns"),
                }
            )
        except Exception:
            continue

    return summaries


def recommendation_to_label(recommendation: str) -> str:
    mapping = {
        "red": "🔴 block",
        "yellow": "🟡 review",
        "green": "🟢 pass",
    }
    return mapping.get(recommendation, recommendation)

### Важно: ``AIRFLOW_UID``

Airflow в Docker работает с локально примонтированными директориями проекта, поэтому ему нужен корректный ``AIRFLOW_UID``.

В файле ``.env`` должно быть указано, например:

```env
AIRFLOW_UID=1000
```

Проверить свой UID можно так:

```bash
id -u
```

Если команда возвращает не ``1000``, замени значение ``AIRFLOW_UID`` в ``.env`` на своё.

Это нужно для того, чтобы:

- контейнеры Airflow корректно запускались;
- логи и артефакты создавались с ожидаемыми правами;
- не возникали ошибки доступа к файлам на хосте;
- не приходилось вручную исправлять владельца файлов через ``chown``.

Если ``AIRFLOW_UID`` задан неверно или не задан вообще, возможны проблемы со стартом Airflow и ошибки прав доступа при записи в ``logs/`, ``data/processed/` и ``artifacts/`.

docker-compose up -d airflow-init airflow-webserver airflow-scheduler

docker-compose ps

docker-compose stop

Остановить и удалить контейнеры:

## 7. Как очистить проект

Удалить артефакты:

```bash
rm -rf artifacts/*
rm -rf data/processed/*
```

Если есть проблемы с правами:

```bash
sudo rm -rf artifacts
mkdir -p artifacts/metrics artifacts/models artifacts/reports artifacts/data artifacts/predictions artifacts/shap
```

Полная очистка контейнерной инфраструктуры:

```bash
docker-compose down -v
```

После этого базы `airflow` и `mlflow` будут создаваться заново.

## 8. Частые проблемы

### Проблема: database `mlflow` does not exist

Проверить список баз:

```bash
docker-compose exec postgres psql -U aml_user -d postgres -c "\l"
```

Если базы нет, значит init-скрипт не выполнился, либо volume PostgreSQL уже существовал до добавления SQL-файла.

Решение: сделать чистый сброс PostgreSQL через `docker-compose down -v` и поднять заново.

---

### Проблема: Airflow пишет `You need to upgrade the database`

Причина: Airflow metadata DB не инициализирована или используется не та база.

Нужно, чтобы Airflow использовал отдельную БД `airflow`, а MLflow — отдельную БД `mlflow`.

После исправления конфигурации нужно поднять инфраструктуру заново.

---

### Проблема: `Permission denied` в `artifacts`

Причина: часть файлов создана контейнером под другим пользователем.

Быстрое решение:

```bash
sudo rm -rf artifacts
mkdir -p artifacts/metrics artifacts/models artifacts/reports artifacts/data artifacts/predictions artifacts/shap
```

---

### Проблема: MLflow долго стартует

Если внутри контейнера каждый раз выполняется `pip install`, запуск может быть медленным.

Это не критическая ошибка, но лучше вынести установку зависимостей в отдельный Dockerfile для сервиса MLflow.

----- FILE: ./dump/dump_project.sh -----
#!/usr/bin/env bash
set -euo pipefail

ROOT_DIR="${1:-$HOME/HSE_AML_MLOps_pipeline}"
OUTPUT_FILE="${2:-project_dump.txt}"
MAX_TEXT_FILE_SIZE="${MAX_TEXT_FILE_SIZE:-524288}" # 512 KB

if [[ ! -d "$ROOT_DIR" ]]; then
  echo "Директория не найдена: $ROOT_DIR" >&2
  exit 1
fi

should_skip_path() {
  local path="$1"

  case "$path" in
    */.git/*|*/.idea/*|*/.vscode/*|*/.venv/*|*/venv/*|*/env/*|*/__pycache__/*|\
*/node_modules/*|*/mlruns/*|*/logs/*|*/artifacts/*|*/postgres_data/*|*/minio_data/*|\
*/.pytest_cache/*|*/.mypy_cache/*|*/.ruff_cache/*|*/.tox/*|*/.cache/*|\
*/dist/*|*/build/*|*/site-packages/*|*/htmlcov/*|*/coverage/*|*/.eggs/*|*/.next/*)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

should_include_file() {
  local file="$1"
  local base lower

  base="$(basename "$file")"
  lower="$(printf '%s' "$base" | tr '[:upper:]' '[:lower:]')"

  case "$lower" in
    dockerfile|makefile|readme|readme.md|license|license.md|requirements.txt|\
pyproject.toml|poetry.lock|pipfile|pipfile.lock|setup.py|setup.cfg|tox.ini|\
.gitignore|.dockerignore|.env|.env.example|docker-compose.yml|docker-compose.yaml)
      return 0
      ;;
  esac

  case "$lower" in
    *.py|*.sh|*.bash|*.zsh|*.md|*.txt|*.yml|*.yaml|*.json|*.toml|*.ini|*.cfg|*.conf|*.sql|*.xml)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

is_text_file() {
  local file="$1"

  if ! command -v file >/dev/null 2>&1; then
    return 0
  fi

  if file --mime "$file" | grep -q 'charset=binary'; then
    return 1
  fi

  return 0
}

get_file_size() {
  local file="$1"

  if stat -c%s "$file" >/dev/null 2>&1; then
    stat -c%s "$file"
    return 0
  fi

  if stat -f%z "$file" >/dev/null 2>&1; then
    stat -f%z "$file"
    return 0
  fi

  wc -c < "$file"
}

contains_embedded_binary_blob() {
  local file="$1"

  if grep -qE 'data:(font|image|application)/[^;]+;base64,' "$file" 2>/dev/null; then
    return 0
  fi

  if awk 'length($0) > 5000 { found=1; exit } END { exit !found }' "$file" 2>/dev/null; then
    return 0
  fi

  return 1
}

relative_path() {
  local file="$1"

  if command -v realpath >/dev/null 2>&1; then
    realpath --relative-to="$ROOT_DIR" "$file" 2>/dev/null && return 0
  fi

  python3 - "$ROOT_DIR" "$file" <<'PY'
import os
import sys
print(os.path.relpath(sys.argv[2], sys.argv[1]))
PY
}

{
  echo "PROJECT ROOT: $ROOT_DIR"
  echo "GENERATED AT: $(date '+%Y-%m-%d %H:%M:%S')"
  echo "MAX TEXT FILE SIZE: $MAX_TEXT_FILE_SIZE bytes"
  echo

  echo "===================="
  echo "DIRECTORY STRUCTURE"
  echo "===================="
  echo

  find "$ROOT_DIR" \
    \( -path "*/.git" -o -path "*/.idea" -o -path "*/.vscode" -o -path "*/.venv" -o -path "*/venv" -o -path "*/env" -o \
       -path "*/__pycache__" -o -path "*/node_modules" -o -path "*/mlruns" -o -path "*/logs" -o -path "*/artifacts" -o \
       -path "*/postgres_data" -o -path "*/minio_data" -o -path "*/.pytest_cache" -o -path "*/.mypy_cache" -o \
       -path "*/.ruff_cache" -o -path "*/.tox" -o -path "*/.cache" -o -path "*/dist" -o -path "*/build" -o \
       -path "*/site-packages" -o -path "*/htmlcov" -o -path "*/coverage" -o -path "*/.eggs" -o -path "*/.next" \) -prune \
    -o -print | sed "s|$ROOT_DIR|.|" | sort

  echo
  echo "================="
  echo "FILES CONTENT"
  echo "================="
  echo

  while IFS= read -r file; do
    if should_skip_path "$file"; then
      continue
    fi

    if [[ ! -f "$file" ]]; then
      continue
    fi

    if ! should_include_file "$file"; then
      continue
    fi

    if ! is_text_file "$file"; then
      continue
    fi

    if [[ "$(get_file_size "$file")" -gt "$MAX_TEXT_FILE_SIZE" ]]; then
      continue
    fi

    if contains_embedded_binary_blob "$file"; then
      continue
    fi

    rel_path="$(relative_path "$file")"

    echo "----- FILE: ./$rel_path -----"
    cat "$file"
    echo
    echo "----- END FILE: ./$rel_path -----"
    echo
  done < <(
    find "$ROOT_DIR" \
      \( -path "*/.git" -o -path "*/.idea" -o -path "*/.vscode" -o -path "*/.venv" -o -path "*/venv" -o -path "*/env" -o \
         -path "*/__pycache__" -o -path "*/node_modules" -o -path "*/mlruns" -o -path "*/logs" -o -path "*/artifacts" -o \
         -path "*/postgres_data" -o -path "*/minio_data" -o -path "*/.pytest_cache" -o -path "*/.mypy_cache" -o \
         -path "*/.ruff_cache" -o -path "*/.tox" -o -path "*/.cache" -o -path "*/dist" -o -path "*/build" -o \
         -path "*/site-packages" -o -path "*/htmlcov" -o -path "*/coverage" -o -path "*/.eggs" -o -path "*/.next" \) -prune \
      -o -type f -print | sort
  )

} > "$OUTPUT_FILE"

echo "Готово: $OUTPUT_FILE"
